def check_environments
  @platform.environments.each do |env|
    puts Elevage::Environment.new(env).healthy? ? (env + MSG_ENV_HEALTHY) : fail(IOError, ERR[:fail_health_check])
  end
end

def check_platform
  @platform = Elevage::Platform.new

end
